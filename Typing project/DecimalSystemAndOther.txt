int hodnotaVeDeci(string str,int b) {
	return stoi(str, 0, b);
}
string hodnotaVeJinem(int num, int base) {
	string new_value;
	int remainder;
	while (num != 0) {
		remainder = num % base;
		num = num / base;
		if (remainder > 9 && remainder < 16) {
			new_value += char(remainder + 55);
		}
		else if (remainder < 9) {
			new_value += to_string(remainder);
		}
	}
	return new_value;
}
