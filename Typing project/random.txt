#include <iostream>  //enter
#include<cstdlib>
#include<ctime>
using namespace std;

int intRan(int min, int max);
float floatRan(float min, float max);

int main()
{//enter+tab
	//kazdy provedeni je jine.
	srand((time(NULL)));

	for (int i = 0; i < 5; ++i) {
		cout << floatRan(10.0, 20.0) << endl;
	}
	for (int i = 0; i < 5; ++i) {
		cout << intRan(10, 20) << endl;
	}
	return 0;
}

int intRan(int min, int max) {
	// m%n is in [0,n-1]
	return min + rand() % (max + 1 - min);
	//rand()% m    return integer number in range [0,m-1]
}

float floatRan(float min, float max) {
	//rand() / float(RAND_MAX) is in [0,1.0]
	return min + (max - min) * (rand() / float(RAND_MAX));
}
